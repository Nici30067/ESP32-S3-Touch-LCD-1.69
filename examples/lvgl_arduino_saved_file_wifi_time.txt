void my_touchpad_read(lv_indev_drv_t *indev_driver, lv_indev_data_t *data) {  //Void get TOUCHPAD data
  int32_t touchX = CST816T->IIC_Read_Device_Value(CST816T->Arduino_IIC_Touch::Value_Information::TOUCH_COORDINATE_X);
  int32_t touchY = CST816T->IIC_Read_Device_Value(CST816T->Arduino_IIC_Touch::Value_Information::TOUCH_COORDINATE_Y);

  if (CST816T->IIC_Interrupt_Flag == true) {
    CST816T->IIC_Interrupt_Flag = false;
    data->state = LV_INDEV_STATE_PR;

    /* Set the coordinates with some debounce */
    if (touchX >= 0 && touchY >= 0) {
      data->point.x = touchX;
      data->point.y = touchY;

      USBSerial.printf("Data x: %d, Data y: %d\n", touchX, touchY);
    }
  } else {
    data->state = LV_INDEV_STATE_REL;
  }
}

static void check_if_power_button_is_pressed_once() {
  //check if power button is pressed
  int reading = digitalRead(inputPin);
  int checking_cur_lcd_state = digitalRead(LCD_BL);
  if (reading != buttonState) {
    buttonState = reading;

    if (buttonState == LOW) {
      // Operation when button is pressed - single click detected

      clickDetected = true;
      //esp_sleep_enable_timer_wakeup(10 * 1000000ULL);
      //esp_sleep_enable_ext0_wakeup(GPIO_NUM_27, level);
      //esp_sleep_enable_ext1_wakeup_io(bitmask, ESP_EXT1_WAKEUP_ANY_LOW);
      //digitalWrite(LCD_BL, LOW);
      //delay(1000);
      //esp_deep_sleep_start();
      if (checking_cur_lcd_state == HIGH) {

        digitalWrite(LCD_BL, LOW);

      } else if (checking_cur_lcd_state == LOW) {
        digitalWrite(LCD_BL, HIGH);
      }
    }
    delay(100);  // Used to eliminate button noise
  }
}
